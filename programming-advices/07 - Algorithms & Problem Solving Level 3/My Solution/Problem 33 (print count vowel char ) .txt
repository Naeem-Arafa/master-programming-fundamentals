//#include <bits/stdc++.h>#include <iostream>#include <iomanip>#include <cstdlib>#include <string>#define ll long longusing namespace std;string ReadString(){	string str = " ";	getline(cin, str);	return str;}bool IsVowel(char ch){	ch = tolower(ch);	if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u')	{		return true;	}	else	{		return false;	}}short CountVowelChar(string str){	short Count = 0;	for (int i = 0; i < str.size(); i++)	{		if (IsVowel(str[i]))		{			Count++;		}	}	return Count;}int main(){	cout << "please enter your string.\n";	string str = ReadString();	cout << "Number of vowels is : " << CountVowelChar(str) << endl;				}
