//#include <bits/stdc++.h>#include <iostream>#include <iomanip>#include <cstdlib>#include <string>#define ll long longusing namespace std;string ReadString(){	string str = " ";	getline(cin, str);	return str;}char ReadChar(){	char str = ' ';	cin >> str;	return str;}char InverseLetterCase(char str){	return isupper(str) ? tolower(str) : toupper(str);}short CountCharInString(string str, char ch){	short Count = 0;	for (int i = 0; i < str.size(); i++)	{		if (str[i] == ch)		{			Count++;		}	}	return Count;}int main(){	cout << "please enter your string.\n";	string str = ReadString();	cout << "please enter a char.\n";	char ch = ReadChar();	cout << "Letter " << ch << " Count = " << CountCharInString(str, ch) << endl;		cout << "Letter " << ch << " Or " << InverseLetterCase(ch) << " Count = " << CountCharInString(str, InverseLetterCase(ch)) + CountCharInString(str, ch) << endl;			}
